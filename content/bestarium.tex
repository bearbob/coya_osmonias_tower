\newcommand{\setMonsterText}{
  \setlength{\parskip}{0em}
  \scriptsize
}
\newcommand{\resetMonsterText}{
  \setlength{\parskip}{1em}
  \large
}

\newcommand{\monsterSpider}[2]{
  \setMonsterText
  \begin{monsterbox}{Riesige Wolfsspinne}
    \begin{hangingpar}
      \textit{Initiative +3}
    \end{hangingpar}
    \dndline%
    \basics[%
    armorclass = 14,
    hitpoints  = #1,
    speed      = {30 ft.}
    ]
    \dndline\stats[
      STR = \stat{14}, % This stat command will autocomplete the modifier for you
      DEX = \stat{16},
      CON = \stat{12},
      INT = \stat{2},
      WIS = \stat{11},
      CHA = \stat{4},
    ]
    \dndline%
    \monsterattack[
      name=Biss,
      type={melee},
      mod=+3,%mod=+0,
      %reach=5,
      %range=20/60,
      %targets=one target,
      dmg=\dice{1d8+3},
      dmgtype=piercing,
      %plusdmg=,
      %plusdmgtype=,
      %ordmg=,
      %ordmgwhen=,
      %extra=,
    ]
    \dndline
    \begin{monsteraction}[Giftig]
      Die Angriffe der Spinne sind giftig. Wirf jedes Mal einen Konstitutionsrettungswurf mit
      DC 14, wenn du getroffen wirst. Bist du erfolgreich, wird der Kampf fortgesetzt.
      Wenn nicht, gehe zu [\ref{#2}].
    \end{monsteraction}
  \end{monsterbox}
  \resetMonsterText
}
